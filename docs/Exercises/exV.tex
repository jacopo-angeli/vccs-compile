% ================================================================
% GOAL
% ================================================================

\noindent\textbf{Goal.}
Define in pure CCS an operator $\tau_L(P)$ satisfying, for a finite set of channel
names $L$:
\[
  \begin{array}{rcll}
    \tau_L(a.P)              & = & \tau.\tau_L(P)              & a \in L              \\
    \tau_L(\bar{a}.P)        & = & \tau.\tau_L(P)              & a \in L              \\
    \tau_L(\alpha.P)         & = & \alpha.\tau_L(P)            & \alpha \notin L \cup \bar{L} \\
    \tau_L(P + Q)            & = & \tau_L(P) + \tau_L(Q)       &                      \\
    \tau_L(P \mid Q)         & = & \tau_L(P) \mid \tau_L(Q)    &                      \\
  \end{array}
\]

% ================================================================
% CONSTRUCTION
% ================================================================

\noindent\textbf{Construction.}
For each $a \in L$, let $b_a$ be a fresh channel name not appearing in $P$ or $L$.
Define the \emph{absorber} process by the recursive CCS equation:
\[
  A_a \;\overset{\text{def}}{=}\; b_a.A_a \;+\; \bar{b}_a.A_a
\]
which persistently offers to synchronize on $b_a$ in either direction and loops back.
Write $A_L = \prod_{a \in L} A_a$ for the parallel composition of all absorbers.
The encoding is:
\[
  \tau_L(P) \;\overset{\text{def}}{=}\;
  \Bigl(\,P\bigl[\,b_a/a \;\mid\; a \in L\,\bigr]
  \;\mid\; A_L\,\Bigr)
  \;\setminus\; \{\,b_a \mid a \in L\,\}
\]

\noindent\textit{Intuition.}
Each channel $a \in L$ is renamed to a fresh $b_a$.
The absorber $A_a$ always offers the complementary action on $b_a$, forcing any
visible move on $b_a$ to immediately synchronize and produce $\tau$.
Restriction hides $b_a$ entirely, so no external observer can see it.

\bigskip

% ================================================================
% VERIFICATION
% ================================================================

\noindent\textbf{Verification} — we check each characteristic equation by exhibiting
the required CCS derivation.

\medskip
\noindent\textbf{IH} (for the inductive cases):
assume $\tau_L(P_0)$ behaves correctly for all strict subprocesses $P_0$.
Write $P_0^* = P_0[b_a/a \mid a \in L]$ for brevity.

\bigskip

\noindent\textbf{Case} $P = a.Q$, $a \in L$:

\noindent By the renaming, $(a.Q)^* = b_a.Q^*$.
The absorber $A_a = b_a.A_a + \bar{b}_a.A_a$ offers $\bar{b}_a$ via its right branch:

\begin{prooftree}
  \AxiomC{}
  \RightLabel{\textsc{Act}}
  \UnaryInfC{$b_a.Q^* \xrightarrow{b_a} Q^*$}
  \AxiomC{}
  \RightLabel{\textsc{Act}}
  \UnaryInfC{$\bar{b}_a.A_a \xrightarrow{\bar{b}_a} A_a$}
  \RightLabel{\textsc{Sum}}
  \UnaryInfC{$A_a \xrightarrow{\bar{b}_a} A_a$}
  \RightLabel{\textsc{Par-L}}
  \UnaryInfC{$A_L \xrightarrow{\bar{b}_a} A_a \mid A_{L\setminus\{a\}}$}
  \RightLabel{\textsc{Com}}
  \BinaryInfC{$b_a.Q^* \mid A_L \xrightarrow{\tau} Q^* \mid A_L$}
  \RightLabel{\textsc{Res} \quad $\tau \notin \{b_a \mid a \in L\}$}
  \UnaryInfC{$(b_a.Q^* \mid A_L)\setminus\{b_a \mid a \in L\} \xrightarrow{\tau}
              (Q^* \mid A_L)\setminus\{b_a \mid a \in L\}$}
\end{prooftree}
The right-hand side is $(Q^* \mid A_L)\setminus\{b_a \mid a \in L\} = \tau_L(Q)$.
Hence $\tau_L(a.Q) \xrightarrow{\tau} \tau_L(Q)$, matching $\tau.\tau_L(Q)$. $\checkmark$

\bigskip

\noindent\textbf{Case} $P = \bar{a}.Q$, $a \in L$:

\noindent Relabeling satisfies $f(\bar{a}) = \overline{f(a)}$, so $(\bar{a}.Q)^* = \bar{b}_a.Q^*$.
The absorber $A_a$ offers $b_a$ via its left branch:

\begin{prooftree}
  \AxiomC{}
  \RightLabel{\textsc{Act}}
  \UnaryInfC{$\bar{b}_a.Q^* \xrightarrow{\bar{b}_a} Q^*$}
  \AxiomC{}
  \RightLabel{\textsc{Act}}
  \UnaryInfC{$b_a.A_a \xrightarrow{b_a} A_a$}
  \RightLabel{\textsc{Sum}}
  \UnaryInfC{$A_a \xrightarrow{b_a} A_a$}
  \RightLabel{\textsc{Par-L}}
  \UnaryInfC{$A_L \xrightarrow{b_a} A_a \mid A_{L\setminus\{a\}}$}
  \RightLabel{\textsc{Com}}
  \BinaryInfC{$\bar{b}_a.Q^* \mid A_L \xrightarrow{\tau} Q^* \mid A_L$}
  \RightLabel{\textsc{Res} \quad $\tau \notin \{b_a \mid a \in L\}$}
  \UnaryInfC{$(\bar{b}_a.Q^* \mid A_L)\setminus\{b_a \mid a \in L\} \xrightarrow{\tau}
              (Q^* \mid A_L)\setminus\{b_a \mid a \in L\}$}
\end{prooftree}
Hence $\tau_L(\bar{a}.Q) \xrightarrow{\tau} \tau_L(Q)$. $\checkmark$

\bigskip

\noindent\textbf{Case} $P = \alpha.Q$, $\alpha \notin L \cup \bar{L}$:

\noindent The relabeling does not touch $\alpha$, so $(\alpha.Q)^* = \alpha.Q^*$.
No absorber can synchronize on $\alpha$ since all $b_a$ are fresh.
The restriction does not block $\alpha$ since $\alpha \notin \{b_a \mid a \in L\}$:

\begin{prooftree}
  \AxiomC{}
  \RightLabel{\textsc{Act}}
  \UnaryInfC{$\alpha.Q^* \xrightarrow{\alpha} Q^*$}
  \RightLabel{\textsc{Par-L}}
  \UnaryInfC{$\alpha.Q^* \mid A_L \xrightarrow{\alpha} Q^* \mid A_L$}
  \RightLabel{\textsc{Res} \quad $\alpha \notin \{b_a \mid a \in L\}$}
  \UnaryInfC{$(\alpha.Q^* \mid A_L)\setminus\{b_a \mid a \in L\} \xrightarrow{\alpha}
              (Q^* \mid A_L)\setminus\{b_a \mid a \in L\}$}
\end{prooftree}
Hence $\tau_L(\alpha.Q) \xrightarrow{\alpha} \tau_L(Q)$. $\checkmark$

\bigskip

\noindent\textbf{Case} $P = P_1 + P_2$:

\noindent $(P_1+P_2)^* = P_1^* + P_2^*$, so any transition of the sum comes from
one component $j \in \{1,2\}$.
By IH on $P_j$, $\tau_L(P_j)$ behaves correctly.
The Sum rule then gives $\tau_L(P_1) + \tau_L(P_2)$ the same transitions:

\begin{prooftree}
  \AxiomC{$P_j^* \mid A_L \xrightarrow{\mu} Q_j \mid A_L$}
  \RightLabel{IH}
  \UnaryInfC{$\tau_L(P_j) \xrightarrow{\mu} \tau_L(P_j')$}
  \RightLabel{\textsc{Sum}}
  \UnaryInfC{$\tau_L(P_1) + \tau_L(P_2) \xrightarrow{\mu} \tau_L(P_j')$}
\end{prooftree}
Hence $\tau_L(P_1 + P_2) = \tau_L(P_1) + \tau_L(P_2)$. $\checkmark$

\bigskip

\noindent\textbf{Case} $P = P_1 \mid P_2$:

\noindent $(P_1 \mid P_2)^* = P_1^* \mid P_2^*$.
Each absorber $A_a$ is independent across components, so
$A_L$ distributes: $P_1^* \mid A_L \mid P_2^*$ rearranges to
$( P_1^* \mid A_L) \mid (P_2^* \mid A_L)$ up to the structural equivalence
$A_L \sim A_L \mid A_L$ (since $A_a$ is replicated by its own recursion).
By IH on each component, Par-L/Par-R/Com then give
$\tau_L(P_1 \mid P_2) = \tau_L(P_1) \mid \tau_L(P_2)$. $\checkmark$

\bigskip

% ================================================================
% CONCLUSION
% ================================================================

\noindent\textbf{Definability in CCS.}
The construction uses only:
\[
  \begin{array}{ll}
    P[b_a/a \mid a \in L]                           & \text{relabeling — a standard CCS operator} \\
    A_a \overset{\text{def}}{=} b_a.A_a + \bar{b}_a.A_a & \text{recursive constant — allowed in CCS}  \\
    A_L = \prod_{a \in L} A_a                        & \text{finite parallel composition}            \\
    (\cdots)\setminus\{b_a \mid a \in L\}            & \text{restriction — a standard CCS operator}  \\
  \end{array}
\]
All operators belong to the standard CCS syntax.
Hence $\tau_L$ is definable in CCS. $\square$

\bigskip

% ================================================================
% FURTHER OBSERVATIONS
% ================================================================

\noindent\textbf{Further observations.}

\medskip
\noindent\textbf{1. Freshness of $b_a$ is essential.}
The construction requires $b_a \notin \mathrm{fn}(P) \cup L$ strictly.
If $b_a$ already appeared in $P$, the process could synchronize with $A_a$
on $b_a$ at unintended moments — not because of an action in $L$, but because
$P$ happened to use the same channel name. The result would be spurious $\tau$
steps that do not correspond to any original action in $L$.

\medskip
\noindent\textbf{2. $\tau_L$ vs.\ restriction $P \setminus L$.}
These are fundamentally different operators.
$P \setminus L$ \emph{blocks} actions in $L$: the process gets stuck wherever
it would need $a \in L$.
$\tau_L(P)$ \emph{absorbs} them: the action still happens internally and the
process continues, but the external observer only sees $\tau$.
In particular, $\tau_L(a.P) \xrightarrow{\tau} \tau_L(P)$ whereas
$(a.P) \setminus \{a\}$ has no transition at all.

\medskip
\noindent\textbf{3. Existing $\tau$ actions are unaffected.}
If $P$ already contains a $\tau$ prefix, it falls under the case
$\alpha \notin L \cup \bar{L}$ (since $\tau$ has no complement in $L$) and
passes through unchanged:
$\tau_L(\tau.P) = \tau.\tau_L(P)$.
The construction does not introduce extra silent steps beyond those caused by
absorbing actions in $L$.

\medskip
\noindent\textbf{4. Why $A_L \sim A_L \mid A_L$.}
The parallel case relies on the absorber being \emph{replicable}: one copy of
$A_a$ can serve arbitrarily many requests because after each synchronization it
loops back to $A_a$ unchanged.
Formally, $A_a$ is weakly bisimilar to $A_a \mid A_a$: both processes always
offer $b_a$ and $\bar{b}_a$, and after any sequence of synchronizations they
return to the same shape.
This is what allows a single $A_L$ in the construction to simultaneously serve
all components of a parallel composition.

\medskip
\noindent\textbf{5. Degenerate case $L = \emptyset$.}
When $L$ is empty there are no channels to rename, no absorbers to add, and no
channels to restrict. The construction reduces to:
\[
  \tau_\emptyset(P) = (P \mid \mathbf{0}) \setminus \emptyset = P
\]
so $\tau_\emptyset$ is the identity, as expected.

\medskip
\noindent\textbf{6. The verification is not a full bisimulation proof.}
The cases above show that $\tau_L(P)$ can \emph{perform} the required
transitions, but a complete proof of behavioural equivalence would also require
showing that $\tau_L(P)$ cannot perform any \emph{extra} transitions — i.e.\
proving the converse direction and closing the bisimulation relation.
The construction ensures this because the restriction on $\{b_a\}$ prevents any
action on those channels from being observable, and the absorber introduces no
new visible actions (it only synchronizes, producing $\tau$).
