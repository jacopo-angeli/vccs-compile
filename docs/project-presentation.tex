\section{Architecture \& Implementation}

  \begin{frame}
    \frametitle{The Key Idea}

    Value passing is encoded as a \textbf{choice of channel name}.

    \vspace{0.5cm}

    \textbf{Input action:}
    $$a(x:[0,10]).P' \;\implies\; a_0.P' + a_1.P' + \ldots + a_{10}.P'$$
    $$a(x:[0,10]).P'(x) \;\implies\; a_0.P_0 + a_1.P_1 + \ldots + a_{10}.P_{10}$$

    \vspace{0.5cm}

    \textbf{Output action}:
    $$\overline{a}(10).P \;\implies\; \overline{a}_{10}.P$$

    \vspace{0.5cm}

    \textbf{Parameterised constants}:
    $$Q(x:[0,5], y:[0,2]) \;\implies\; Q_{0.0},\; Q_{0.1},\; \ldots,\; Q_{5.2}$$
  \end{frame}

  \begin{frame}
    \frametitle{Project Structure}

    \begin{itemize}
      \item \texttt{src/Program.fs} — CLI entry point; file mode and interactive REPL
      \item \texttt{src/Compiler.fs} — Translation logic from VCCS to PCCS
      \item \texttt{src/Vccs/} — source-language AST, lexer and parser (fslex/fsyacc)
      \item \texttt{src/Pccs/} — target-language AST and pretty-printer
      \item \texttt{test/} — example inputs and expected outputs
    \end{itemize}
  \end{frame}

  \begin{frame}[fragile]
    \frametitle{Environment}

    \begin{columns}[t]
      \column{0.5\textwidth}
      \textbf{Built} on each input action:
      \begin{lstlisting}
compileP cont
  ((var, value) :: env)
      \end{lstlisting}
      One copy of \texttt{env} per value in \texttt{[L,H]}.

      \column{0.5\textwidth}
      \textbf{Read} when evaluating expressions:
      \begin{lstlisting}
| Var x ->
    List.find
      (fun (n,_) -> n = x)
      env |> snd
      \end{lstlisting}
      Closure over \texttt{compileP}'s \texttt{env}.
    \end{columns}

  \end{frame}
